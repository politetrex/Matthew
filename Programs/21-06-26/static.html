AIO 2021 P1 - Robot Vacuum
Core Idea: Record it's $D_x$ and $D_y$ (X distance and Y distance).
Initiate $D_x$ and $D_y$ to $0$. Then, you iterate through the command line with this:
  • N: $D_y$++;
  • E: $D_x$++;
  • S: $D_y$--;
  • W: $D_x$--.
Source Code 
#include <iostream>
using namespace std;

void p1() {
    freopen("robotin.txt", "r", stdin);
    freopen("robotout.txt", "w", stdout);
    
    int len;
    cin >> len;
    
    int horizontal = 0, vertical = 0;
    
    for (int i = 0; i < len; i++) {
        char ch;
        cin >> ch;
        
        switch (ch) {
            case 'E': horizontal++; break;
            case 'W': horizontal--; break;
            case 'N': vertical++; break;
            case 'S': vertical--; break;
        }
    }
    
    cout << (abs(horizontal) + abs(vertical)) << endl;
}

int main() {
    p1();
    return 0;
}
AIO 2021 P2 - Art Class II
Core Idea: Set four variables $min_x, min_y, max_x, max_y$.
They correspond to the minimum x, minimum y, maximum x, maximum y for the holes. Then, output $(max_x-min_x)*(max_y-min_y)$.
Source Code 
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    
    int minx = 0x3fffffff, maxx = 0;
    int miny = 0x3fffffff, maxy = 0;
    
    while (n--) {
        int x, y;
        cin >> x >> y;
        
        minx = min(minx, x);
        maxx = max(maxx, x);
        miny = min(miny, y);
        maxy = max(maxy, y);
    }
    
    int area = (maxx - minx) * (maxy - miny);
    cout << area;
    
    return 0;
}
AIO 2021 P3 - Melody
Core Idea: Since melodies repeat every 3 notes, we can partition the string into sections of 3.
Then, count the frequencies of each note on each expected note place ($P_0$, $P_1$ and $P_2$, repeats, like: $P_0$, $P_1$, $P_2$, $P_0$, $P_1$, $P_2$, $P_0$, $P_1$, $P_2$, ...)
Sum up the most frequent on each place, then subtract this number from $N$.
Source Code 
#include <iostream>
#include <vector>
using namespace std;
int main(){
    int n, k, tot=0; cin>>n>>k;
    vector<vector<int>> mfreq(3,vector<int>(k+1,0));
    for(int i=0;i<n;i++){
        int x; cin>>x;
        mfreq[i%3][x]++;
    }
    for(int i=0;i<3;i++){
        int maxn=-1;
        for(int j=1;j<=k;j++){
            maxn=max(maxn,mfreq[i][j]);
        }
        tot+=maxn;
    }
    cout<<n-tot; return 0;
}
AIO 2021 P4 - Social Distancing
Core Idea: The problem is a greedy problem. Why?
First, sort the meals in ascending order.
Then, from the nearest meal, find the first meal such that it's distance from the meal we've got $ ≥ K $.
Repeat until you reached the end of the sorted meals.
Source Code 
#include <iostream>
#include <vector>
#include <algorithm>
 
using namespace std;
 
int main() {
    cin.tie(0)->sync_with_stdio(0);
    int n, k;
    cin>>n>>k;
    vector<int> a(n);
    for(int i=0; i<n; i++)cin>>a[i];
    sort(a.begin(), a.end());
 
    int count=0, prev=0-k;
    for(int i=0; i<n; i++){
        if(a[i]-prev>=k){
            count++;
            prev=a[i];
        }
    }
 
    cout<<count<<endl;
 
    return 0;
}
AIO 2021 P5 - Space Mission Since the original problem statement is a bit o' long, it's actually a short problem:
Finding a pair of indices which is farthest and their corresponding fuel cost add up ≤ f
Core Idea: For each possible return day $j$, we want to find the earliest launch day $i<j$ such that $C_i+C_j≤F$ to maximize samples $j - i + 1$.
$minp_i$=minimum cost among days $0$ to $i$ (non-increasing), allowing us to binary search for the first index where $minp_{mid}≤F-C_j$.
The first valid index lo found gives the earliest affordable launch day for return day $j$, and we take the maximum $j-lo+1$ across all $j$, outputting $-1$ if no valid pair exists.
Source Code 
#include <iostream>
#include <vector>

using namespace std;

int main() {
    cin.tie(0)->sync_with_stdio(0);
    int n,f; cin>>n>>f;
    vector<int> c(n); for(auto& i:c) cin>>i;
    vector<long long> minp(n);
    minp[0] = c[0];
    for (int i=1; i<n; i++)
        minp[i]=min(minp[i-1], (long long)c[i]);
    int ans=-1;
    for (int j=0; j<n; j++) {
        long long maxc=f-c[j];
        if (maxc<=0) continue;
        int lo=0, hi=j; // bSearch.
        while (lo<hi) {
            int mid=(lo+hi)/2;
            if (minp[mid]<=maxc)
                hi=mid;
            else lo=mid+1;
        }
        if (lo<j&&minp[lo]<=maxc)
            ans=max(ans, j-lo+1);
    }
    cout<<ans; return 0;
}
